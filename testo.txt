------FUNZIONI------

funzione RiscaldaIlLatte() {
    prendi un pentolino;
    versa il latte nel pentolino;
    accendi il fuoco;
    metti sul fuoco il pentolino;
    finche il latte non è caldo:
        prendi il cucchiaio;
        assaggia;
    bevi;
}

funzione CercaIlLatte() {
    controlla se c'è il latte in frigo:
    se c'è:
        prendilo;
        funzione RiscaldaIlLatte:
    se non c'è: 
        oggi sei un respiriano; 
}
